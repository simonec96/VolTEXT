static PDDocument doc;
static PDPage page;

    public static void main(String[] args) throws IOException
    {
    	
        	Prova_Scarpe_div2 creator = new Prova_Scarpe_div2();
        	String url="./Example_pdf/fileScarpe_div4.pdf";
        	doc=new PDDocument();
        	page=new PDPage(PDRectangle.A4);
        	doc.addPage(page);
        	//doc.save(url);
           	creator.add_image_spaced("./resources/logoUNIBG.jpg",20,20,30);
               	doc.save(url);
    }
    
   
	private void add_image_spaced(String file, int i, int posx, int posy) {
		
			PDImageXObject pdImage;
			try {
				pdImage = PDImageXObject.createFromFile(file, doc);
				//int iw = pdImage.getWidth();
				//int ih = pdImage.getHeight();
				PDPageContentStream cont = new PDPageContentStream(doc, page,PDPageContentStream.AppendMode.APPEND, false);
				//drawImage(immagine in PDImageXObject, puntoXInizio da bottom_left,puntoYInizio_da bottom_left,larghezzaImg,AltezzaImg)
				PDRectangle mediaBox=page.getMediaBox();
				float	dimx=UnitConverter.convPointmm(pdImage.getWidth());
				float	dimy=UnitConverter.convPointmm(pdImage.getHeight());
					
				AffineTransform at=new AffineTransform(dimx,0,0,dimy,UnitConverter.convmmPoint((float)posx),mediaBox.getHeight()-dimy-UnitConverter.convmmPoint((float)posy));
                              
				//System.out.println(at.toString());
				cont.drawImage(pdImage, new Matrix(at));
				cont.close();
               
				
			} catch (IOException e) {
				// TODO Auto-generated catch block
				e.printStackTrace();
			}
            
            
        }
